设计模式

1、单例模式

var xiaowang = (function (a,b) { var aa= function (a) { console.log(a); }; var bb=function (b) { console.log(b); }; return { aa:aa, bb:bb }; })(); xiaowang.aa('aa'); xiaowang.bb('bb');

2、构造函数模式

function Animal(name, color) { this.name = name; this.color = color; this.getName = function () { return this.name; } } //实例一个对象 var cat = new Animal('猫', "白色"); console.log(cat.getName());

3、工厂模式

function Animal1(opts) { var obj = new Object(); obj.name = opts.name; obj.color = opts.color; obj.getInfo = () => console.log(`名称:${obj.name},颜色:${obj.color}`); //箭头函数、模板字符串 return obj; } var cat1 = new Animal1({name: '波斯猫', color: '黑色'}); cat1.getInfo();

4、模块模式=>封装大部分代码，只暴露必需接口

var Car = (function () { var name = '法拉利'; function sayName() { console.log(name); } function getColor(name) { console.log(name); } return { name: sayName, color: getColor } })(); Car.name(); Car.color('绿色');

5、混合模式=>原型模式+构造函数模式

function Animal2(name, color) { this.name = name; this.color = color; console.log(this.name + this.color); } Animal2.prototype.getInfo = function () { console.log('名称:' + this.name); }; var cat2 = new Animal2('狗', '白色'); cat2.getInfo();
